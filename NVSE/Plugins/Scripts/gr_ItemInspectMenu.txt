; "*ItemInspectMenu"
; 0 - Transform
; 1 - Show once flag
; 2 - Description (Default)
; 3 - Description (First)

; Load JSONs
int iSize
int iCount
ref rItem
array_var aJSON
array_var aForE
array_var aTemp
string_var sJSONPath

; Item profiles
array_var aFiles = GetFiles "ItemInspectMenu" "*.json"
if eval (iSize = ar_Size aFiles) > 0

	aFiles = Ar_Sort aFiles 1

	while (iSize -= 1) > -1
		; Validate JSON
		sJSONPath = sv_Construct "Data\ItemInspectMenu\%z" (aFiles[iSize])
		if eval TestExpr (aJSON = ReadFromJSON (sJSONPath))
		else
			PrintC "Item Inspect Menu: Invalid -> %z" sJSONPath
			continue
		endif

		; Add to cache array
		if eval (Ar_Size aJSON) > 0 && (TypeOf aJSON) == "StringMap"
			foreach aForE <- aJSON

				; Check valid form
				if eval IsFormValid (rItem = EditorIDToFormID (aForE["key"])) == 0 || IsBaseForm rItem == 0
					PrintC "Invalid: %z" (aForE["key"])
					continue
				endif
				aTemp = *aForE

				; Inspect blacklist
				if eval ar_HasKey aTemp "blacklist"
					AssignKeyword rItem "ItemInspectMenuBlacklist"
					continue
				endif

				; Transform
				if eval ar_HasKey aTemp "transform"
					AuxVarSetStr "*ItemInspectMenu" (aTemp["transform"]) 0 rItem
				else
					AuxVarSetFlt "*ItemInspectMenu" 0 0 rItem
				endif

				; Show once flag
				AuxVarSetFlt "*ItemInspectMenu" (eval ar_HasKey aTemp "showOnce" && aTemp["showOnce"]) 1 rItem

				; Description
				if eval ar_HasKey aTemp "description" && ar_HasKey aTemp["description"] "default"
					; Default
					AuxVarSetStr "*ItemInspectMenu" (aTemp["description"]["default"]) 2 rItem
					; First pick-up
					if eval ar_HasKey aTemp["description"] "first"
						AuxVarSetStr "*ItemInspectMenu" (aTemp["description"]["first"]) 3 rItem
						if eval (rItem = EditorIDToFormID (aTemp["description"]["first"]))
							SetMessageDisabled rItem 1
						endif
					endif
				endif

				iCount += 1

			loop
		endif
	loop

	PrintC "Item Inspect Menu: Loaded %g item profiles." iCount

endif

; Item transforms
aFiles = GetFiles "ItemInspectMenu\Transforms" "*.json"
aTemp = ar_Null
aTemp = ar_Construct "stringmap"
if eval (iSize = ar_Size aFiles) > 0

	aFiles = Ar_Sort aFiles 1

	while (iSize -= 1) > -1
		; Validate JSON
		sJSONPath = sv_Construct "Data\ItemInspectMenu\Transforms\%z" (aFiles[iSize])
		if eval TestExpr (aJSON = ReadFromJSON (sJSONPath))
		else
			PrintC "Item Inspect Menu: Invalid -> %z" sJSONPath
			continue
		endif
		; Add to cache array
		if eval (Ar_Size aJSON) > 0 && (TypeOf aJSON) == "StringMap"
			ar_Cat aTemp aJSON 1
		endif
	loop

	WriteToJSON aTemp "Data\ItemInspectMenu\_Cache\Transforms.json" "" 0 1 0 1
	PrintC "Item Inspect Menu: Loaded %g item transforms." (Ar_Size aTemp)

endif

aJSON = aForE = aTemp = aFiles = ar_Null
sv_Destruct sJSONPath
