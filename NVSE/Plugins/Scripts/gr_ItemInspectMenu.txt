; "*InspectMenu"
; 0 - Show once flag
; 1 - Description (Default)
; 2 - Description (First)
; 3 -> 5 - Rotation override

; Load override configs
int iIndex = 0
int iSize
ref rItem
ref rMessage
array_var aJSON
array_var aTransformJSON
string_var sJSONPath
string_var sTemp
array_var aTransformMap = ar_Construct "stringmap"
array_var aCategories = ar_Construct "array"
array_var aFiles = GetFiles "ItemInspectMenu" "*.json"
if eval (iSize = ar_Size aFiles) > 0
	PrintC "Item Inspect Menu: Loading JSON files..."
	aFiles = Ar_Sort aFiles 1

	while (iSize -= 1) > -1
		sTemp = aFiles[iSize]
		sJSONPath = sv_Construct "Data\ItemInspectMenu\%z" sTemp

		; Check valid form
		if eval IsFormValid (rItem = EditorIDToFormID (sTemp[0:-6])) == 0
			;PrintC "Item Inspect Menu: Skipping -> %z" (sTemp[0:-6])
			continue
		endif

		; Check valid JSON
		if eval TestExpr (aJSON = ReadFromJSON (sJSONPath))
		else
			PrintC "Item Inspect Menu: Invalid -> %z" sJSONPath
			continue
		endif

		; Show once
		AuxVarSetFlt "*InspectMenu" (eval ar_HasKey aJSON "showOnce" && aJSON["showOnce"]) 0 rItem

		; Description
		if eval ar_HasKey aJSON "description" && ar_HasKey aJSON["description"] "default"

			; Default
			sTemp = aJSON["description"]["default"]
			if eval IsFormValid (rMessage = EditorIDToFormID (sTemp)) && (GetType rMessage == 98 || GetType rMessage == 103)
				sTemp = GetFormDescription rMessage
			endif
			AuxVarSetStr "*InspectMenu" (sTemp) 1 rItem

			; First pick-up
			if eval ar_HasKey aJSON["description"] "first"
				sTemp = aJSON["description"]["first"]
				if eval IsFormValid (rMessage = EditorIDToFormID (sTemp)) && (GetType rMessage == 98 || GetType rMessage == 103)
					sTemp = GetFormDescription rMessage
				endif
			endif
			AuxVarSetStr "*InspectMenu" (sTemp) 2 rItem

		else
			AuxVarSetFlt "*InspectMenu" 0 1 rItem
			AuxVarSetFlt "*InspectMenu"	0 2 rItem
		endif

		; Starting transform
		if eval ar_HasKey aJSON "transform"

			; Load transform from JSON
			sTemp = aJSON["transform"]
			if eval ar_HasKey aTransformMap (sTemp) == 0
				aTransformMap[(sTemp)] = ar_Construct "stringmap"
				aTransformMap[(sTemp)]["rotation"] = ar_Construct "stringmap"
				sJSONPath = ("ItemInspectMenu\Transforms\" + (sTemp) + ".json")

				; Missing JSON (Set default values)
				if eval FileExists (sJSONPath) == 0
					PrintC "Item Inspect Menu: Missing Transform JSON -> %z" ((sTemp) + ".json")
					aTransformMap[(sTemp)]["rotation"]["x"] = 0
					aTransformMap[(sTemp)]["rotation"]["y"] = 0
					aTransformMap[(sTemp)]["rotation"]["z"] = 0
				else
					; Check valid JSON
					sJSONPath = sv_Construct "Data\%z" sJSONPath
					if eval TestExpr (aTransformJSON = ReadFromJSON (sJSONPath))
					else
						PrintC "Item Inspect Menu: Invalid -> %z" sJSONPath
						continue
					endif
					aTransformMap[(sTemp)]["rotation"]["x"] = (aTransformJSON["rotation"]["x"])
					aTransformMap[(sTemp)]["rotation"]["y"] = (aTransformJSON["rotation"]["y"])
					aTransformMap[(sTemp)]["rotation"]["z"] = (aTransformJSON["rotation"]["z"])
				endif

			endif

			; Set rotation from map
			AuxVarSetFlt "*InspectMenu" (aTransformMap[(sTemp)]["rotation"]["x"]) 3 rItem
			AuxVarSetFlt "*InspectMenu" (aTransformMap[(sTemp)]["rotation"]["y"]) 4 rItem
			AuxVarSetFlt "*InspectMenu" (aTransformMap[(sTemp)]["rotation"]["z"]) 5 rItem
		endif

		iIndex += 1

	loop

	PrintC "Item Inspect Menu: Loaded %g form profiles." iIndex

endif

aCategories = aFiles = aJSON = aTransformJSON = aTransformMap = ar_Null
sv_Destruct sJSONPath sTemp
